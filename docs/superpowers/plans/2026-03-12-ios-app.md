# iOS App Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a native UIKit iOS app for Family Hub backed by extended Go REST API endpoints, with OIDC/PKCE authentication via Authentik.

**Architecture:** Feature-by-feature delivery. Backend API endpoints are added first (Chunk 1), then the iOS app is built against them (Chunks 2–4). The iOS app lives in `ios/` inside the monorepo. Each feature follows ViewModel → APIClient → URLSession, with ViewModels tested against a fake APIClient.

**Tech Stack:** Go 1.22 (backend), Swift 5.9+, UIKit, ASWebAuthenticationSession (OIDC/PKCE), URLSession, Keychain Services, XCTest.

```go
func (handler *APIHandler) CompleteChore(w http.ResponseWriter, r *http.Request) {
	ctx := r.Context()
	user := middleware.GetUser(ctx)
	choreID := chi.URLParam(r, "id")

	if err := handler.choreService.CompleteChore(ctx, choreID, user.ID); err != nil {
		writeJSON(w, http.StatusInternalServerError, map[string]string{"error": "failed to complete chore"})
		return
	}

	w.WriteHeader(http.StatusNoContent)
}
```

```go
func (handler *APIHandler) ListMeals(w http.ResponseWriter, r *http.Request) {
	ctx := r.Context()

	weekParam := r.URL.Query().Get("week")
	if weekParam == "" {
		weekParam = time.Now().Format("2006-01-02")
	}

	weekStart, err := time.Parse("2006-01-02", weekParam)
	if err != nil {
		writeJSON(w, http.StatusBadRequest, map[string]string{"error": "invalid week format, use YYYY-MM-DD"})
		return
	}

	weekEnd := weekStart.AddDate(0, 0, 6)

	meals, err := handler.mealPlanRepo.FindAll(ctx, repository.MealPlanFilter{
		DateFrom: weekStart.Format("2006-01-02"),
		DateTo:   weekEnd.Format("2006-01-02"),
	})
	if err != nil {
		writeJSON(w, http.StatusInternalServerError, map[string]string{"error": "failed to load meals"})
		return
	}

	writeJSON(w, http.StatusOK, meals)
}
```

```go
func (handler *APIHandler) ListCalendar(w http.ResponseWriter, r *http.Request) {
	ctx := r.Context()

	monthParam := r.URL.Query().Get("month")
	if monthParam == "" {
		monthParam = time.Now().Format("2006-01")
	}

	monthStart, err := time.Parse("2006-01", monthParam)
	if err != nil {
		writeJSON(w, http.StatusBadRequest, map[string]string{"error": "invalid month format, use YYYY-MM"})
		return
	}

	monthEnd := monthStart.AddDate(0, 1, -1)

	chores, err := handler.choreRepo.FindAll(ctx, repository.ChoreFilter{
		DueAfter:  &monthStart,
		DueBefore: &monthEnd,
	})
	if err != nil {
		writeJSON(w, http.StatusInternalServerError, map[string]string{"error": "failed to load chores"})
		return
	}

	if chores == nil {
		chores = []models.Chore{}
	}

	writeJSON(w, http.StatusOK, map[string]interface{}{
		"chores": chores,
	})
}
```

```swift
// ios/FamilyHub/Auth/KeychainStore.swift

import Foundation
import Security

enum KeychainStore {
    private static let accessTokenKey = "familyhub.accessToken"
    private static let refreshTokenKey = "familyhub.refreshToken"

    static func saveAccessToken(_ token: String) {
        save(key: accessTokenKey, value: token)
    }

    static func loadAccessToken() -> String? {
        load(key: accessTokenKey)
    }

    static func saveRefreshToken(_ token: String) {
        save(key: refreshTokenKey, value: token)
    }

    static func loadRefreshToken() -> String? {
        load(key: refreshTokenKey)
    }

    static func clearAll() {
        delete(key: accessTokenKey)
        delete(key: refreshTokenKey)
    }

    private static func save(key: String, value: String) {
        let data = Data(value.utf8)
        let query: [CFString: Any] = [
            kSecClass: kSecClassGenericPassword,
            kSecAttrAccount: key,
            kSecValueData: data,
            kSecAttrAccessible: kSecAttrAccessibleAfterFirstUnlock
        ]
        SecItemDelete(query as CFDictionary)
        SecItemAdd(query as CFDictionary, nil)
    }

    private static func load(key: String) -> String? {
        let query: [CFString: Any] = [
            kSecClass: kSecClassGenericPassword,
            kSecAttrAccount: key,
            kSecReturnData: true,
            kSecMatchLimit: kSecMatchLimitOne
        ]
        var result: AnyObject?
        let status = SecItemCopyMatching(query as CFDictionary, &result)
        guard status == errSecSuccess, let data = result as? Data else { return nil }
        return String(data: data, encoding: .utf8)
    }

    private static func delete(key: String) {
        let query: [CFString: Any] = [
            kSecClass: kSecClassGenericPassword,
            kSecAttrAccount: key
        ]
        SecItemDelete(query as CFDictionary)
    }
}
```

```swift
// ios/FamilyHub/Networking/APIClient.swift

import Foundation

final class APIClient: APIClientProtocol {
    private let baseURL: URL
    private let session: URLSession
    private let accessToken: () -> String?
    private let onUnauthorized: () -> Void

    init(baseURL: URL, session: URLSession = .shared, accessToken: @escaping () -> String?, onUnauthorized: @escaping () -> Void) {
        self.baseURL = baseURL
        self.session = session
        self.accessToken = accessToken
        self.onUnauthorized = onUnauthorized
    }

    func fetchDashboard() async -> Result<DashboardStats, APIError> {
        await get(path: "/api/dashboard")
    }

    func fetchChores(status: String? = nil) async -> Result<[Chore], APIError> {
        var path = "/api/chores"
        if let status = status { path += "?status=\(status)" }
        return await get(path: path)
    }

    func completeChore(id: String) async -> Result<Void, APIError> {
        await postEmpty(path: "/api/chores/\(id)/complete")
    }

    func fetchMeals(week: String) async -> Result<[MealPlan], APIError> {
        await get(path: "/api/meals?week=\(week)")
    }

    func fetchRecipes() async -> Result<[Recipe], APIError> {
        await get(path: "/api/recipes")
    }

    func fetchRecipe(id: String) async -> Result<Recipe, APIError> {
        await get(path: "/api/recipes/\(id)")
    }

    func fetchCalendar(month: String) async -> Result<CalendarResponse, APIError> {
        await get(path: "/api/calendar?month=\(month)")
    }

    // MARK: - Private helpers

    private func get<T: Decodable>(path: String) async -> Result<T, APIError> {
        await request(path: path, method: "GET")
    }

    private func post<T: Decodable>(path: String) async -> Result<T, APIError> {
        await request(path: path, method: "POST")
    }

    private func postEmpty(path: String) async -> Result<Void, APIError> {
        guard let url = URL(string: path, relativeTo: baseURL) else {
            return .failure(.network("Invalid URL: \(path)"))
        }
        var urlRequest = URLRequest(url: url)
        urlRequest.httpMethod = "POST"
        if let token = accessToken() {
            urlRequest.setValue("Bearer \(token)", forHTTPHeaderField: "Authorization")
        }
        do {
            let (_, response) = try await session.data(for: urlRequest)
            guard let httpResponse = response as? HTTPURLResponse else {
                return .failure(.network("Non-HTTP response"))
            }
            switch httpResponse.statusCode {
            case 200...299: return .success(())
            case 401: onUnauthorized(); return .failure(.unauthorized)
            default: return .failure(.server(httpResponse.statusCode))
            }
        } catch {
            return .failure(.network(error.localizedDescription))
        }
    }

    private func request<T: Decodable>(path: String, method: String) async -> Result<T, APIError> {
        guard let url = URL(string: path, relativeTo: baseURL) else {
            return .failure(.network("Invalid URL: \(path)"))
        }

        var urlRequest = URLRequest(url: url)
        urlRequest.httpMethod = method

        if let token = accessToken() {
            urlRequest.setValue("Bearer \(token)", forHTTPHeaderField: "Authorization")
        }

        do {
            let (data, response) = try await session.data(for: urlRequest)

            guard let httpResponse = response as? HTTPURLResponse else {
                return .failure(.network("Non-HTTP response"))
            }

            switch httpResponse.statusCode {
            case 200...299:
                do {
                    let decoded = try JSONDecoder().decode(T.self, from: data)
                    return .success(decoded)
                } catch {
                    return .failure(.decoding(error.localizedDescription))
                }
            case 401:
                onUnauthorized()
                return .failure(.unauthorized)
            case 404:
                return .failure(.notFound)
            default:
                return .failure(.server(httpResponse.statusCode))
            }
        } catch {
            return .failure(.network(error.localizedDescription))
        }
    }
}
```

```swift
import UIKit

final class DashboardViewController: UIViewController {
    private let viewModel: DashboardViewModel

    private let dueTodayLabel = UILabel()
    private let overdueLabel = UILabel()
    private let tableView = UITableView()
    private var chores: [Chore] = []

    init(viewModel: DashboardViewModel) {
        self.viewModel = viewModel
        super.init(nibName: nil, bundle: nil)
        title = "Home"
    }

    required init?(coder: NSCoder) { fatalError() }

    override func viewDidLoad() {
        super.viewDidLoad()
        view.backgroundColor = .systemBackground
        setupLayout()
        bindViewModel()
        Task { await viewModel.load() }
    }

    private func bindViewModel() {
        viewModel.onStateChange = { [weak self] state in
            guard let self else { return }
            switch state {
            case .loaded(let stats):
                self.dueTodayLabel.text = "Due Today: \(stats.choresDueToday)"
                self.overdueLabel.text = "Overdue: \(stats.choresOverdue)"
                self.chores = stats.choresDueTodayList + stats.choresOverdueList
                self.tableView.reloadData()
            case .failed:
                self.dueTodayLabel.text = "Failed to load"
            default:
                break
            }
        }
    }

    private func setupLayout() {
        // Stat cards row
        let statsStack = UIStackView(arrangedSubviews: [dueTodayLabel, overdueLabel])
        statsStack.axis = .horizontal
        statsStack.distribution = .fillEqually
        statsStack.spacing = 12

        dueTodayLabel.backgroundColor = .systemBlue.withAlphaComponent(0.1)
        dueTodayLabel.textAlignment = .center
        dueTodayLabel.layer.cornerRadius = 8
        dueTodayLabel.clipsToBounds = true

        overdueLabel.backgroundColor = .systemRed.withAlphaComponent(0.1)
        overdueLabel.textAlignment = .center
        overdueLabel.layer.cornerRadius = 8
```

```swift
import Foundation

struct MealDay {
    let date: String
    var meals: [MealPlan]

@MainActor
final class MealsViewModel {
    private let apiClient: APIClientProtocol

    private(set) var days: [MealDay] = []
    var currentWeek: String = ""

    var state: ViewState<[MealDay]> = .idle {
        didSet { onStateChange?(state) }
    }
    var onStateChange: ((ViewState<[MealDay]>) -> Void)?

    init(apiClient: APIClientProtocol) {
        self.apiClient = apiClient
    }

    func load(week: String) async {
        currentWeek = week
        state = .loading
        let result = await apiClient.fetchMeals(week: week)
        switch result {
        case .success(let meals):
            let grouped = Dictionary(grouping: meals, by: \.date)
            days = grouped.keys.sorted().map { date in
                MealDay(date: date, meals: grouped[date] ?? [])
            }
            state = .loaded(days)
        case .failure(let error):
            state = .failed(error)
        }
    }

    func mondayOfCurrentWeek() -> String {
        var calendar = Calendar(identifier: .iso8601)
        calendar.firstWeekday = 2
        let today = Date()
        let monday = calendar.date(from: calendar.dateComponents([.yearForWeekOfYear, .weekOfYear], from: today))!
        return DateFormatter.isoDate.string(from: monday)
    }
}

private extension DateFormatter {
    static let isoDate: DateFormatter = {
        let formatter = DateFormatter()
        formatter.dateFormat = "yyyy-MM-dd"
        return formatter
    }()
}
```

```swift
// ios/FamilyHub/Features/Calendar/CalendarViewModel.swift

import Foundation

@MainActor
final class CalendarViewModel {
    private let apiClient: APIClientProtocol
    private var choresByDate: [String: [Chore]] = [:]

    var state: ViewState<CalendarResponse> = .idle {
        didSet { onStateChange?(state) }
    }
    var onStateChange: ((ViewState<CalendarResponse>) -> Void)?
    var currentMonth: String = ""

    init(apiClient: APIClientProtocol) {
        self.apiClient = apiClient
    }

    func load(month: String) async {
        currentMonth = month
        state = .loading
        let result = await apiClient.fetchCalendar(month: month)
        switch result {
        case .success(let response):
            choresByDate = Dictionary(grouping: response.chores) { $0.dueDate ?? "" }
            state = .loaded(response)
        case .failure(let error):
            state = .failed(error)
        }
    }

    func chores(for date: String) -> [Chore] {
        choresByDate[date] ?? []
    }

    func currentMonthString() -> String {
        let formatter = DateFormatter()
        formatter.dateFormat = "yyyy-MM"
        return formatter.string(from: Date())
    }
}
```

```swift
// ios/FamilyHub/Features/Calendar/CalendarViewController.swift

import UIKit

final class CalendarViewController: UIViewController {
    private let viewModel: CalendarViewModel
    private var selectedDate: String?
    private var daysInMonth: [String] = []
    private let collectionView: UICollectionView = {
        let layout = UICollectionViewFlowLayout()
        layout.minimumInteritemSpacing = 2
        layout.minimumLineSpacing = 2
        let cv = UICollectionView(frame: .zero, collectionViewLayout: layout)
        cv.backgroundColor = .systemBackground
        return cv
    }()
    private let agendaTableView = UITableView()
    private var agendaChores: [Chore] = []

    init(viewModel: CalendarViewModel) {
        self.viewModel = viewModel
        super.init(nibName: nil, bundle: nil)
        title = "Calendar"
    }

    required init?(coder: NSCoder) { fatalError() }

    override func viewDidLoad() {
        super.viewDidLoad()
        view.backgroundColor = .systemBackground
        navigationItem.leftBarButtonItem = UIBarButtonItem(title: "<", style: .plain, target: self, action: #selector(previousMonth))
        navigationItem.rightBarButtonItem = UIBarButtonItem(title: ">", style: .plain, target: self, action: #selector(nextMonth))

        setupCollectionView()
        setupAgendaTable()
        bindViewModel()
        Task { await viewModel.load(month: viewModel.currentMonthString()) }
    }

    private func bindViewModel() {
        viewModel.onStateChange = { [weak self] state in
            guard let self, case .loaded = state else { return }
            self.daysInMonth = self.computeDaysInMonth(self.viewModel.currentMonth)
            self.collectionView.reloadData()
        }
    }

    private func computeDaysInMonth(_ month: String) -> [String] {
        let formatter = DateFormatter()
        formatter.dateFormat = "yyyy-MM"
        guard let date = formatter.date(from: month) else { return [] }
        let calendar = Calendar.current
        let range = calendar.range(of: .day, in: .month, for: date)!
        let dayFormatter = DateFormatter()
        dayFormatter.dateFormat = "yyyy-MM-dd"
        return range.map { day -> String in
            var components = calendar.dateComponents([.year, .month], from: date)
            components.day = day
            return dayFormatter.string(from: calendar.date(from: components)!)
        }
    }

    private func setupCollectionView() {
        collectionView.dataSource = self
        collectionView.delegate = self
        collectionView.register(CalendarDayCell.self, forCellWithReuseIdentifier: "DayCell")
        collectionView.translatesAutoresizingMaskIntoConstraints = false
        view.addSubview(collectionView)
        NSLayoutConstraint.activate([
            collectionView.topAnchor.constraint(equalTo: view.safeAreaLayoutGuide.topAnchor),
            collectionView.leadingAnchor.constraint(equalTo: view.leadingAnchor),
            collectionView.trailingAnchor.constraint(equalTo: view.trailingAnchor),
            collectionView.heightAnchor.constraint(equalToConstant: 280),
        ])
    }

    private func setupAgendaTable() {
        agendaTableView.dataSource = self
        agendaTableView.register(UITableViewCell.self, forCellReuseIdentifier: "AgendaCell")
        agendaTableView.translatesAutoresizingMaskIntoConstraints = false
        view.addSubview(agendaTableView)
        NSLayoutConstraint.activate([
            agendaTableView.topAnchor.constraint(equalTo: collectionView.bottomAnchor),
            agendaTableView.leadingAnchor.constraint(equalTo: view.leadingAnchor),
            agendaTableView.trailingAnchor.constraint(equalTo: view.trailingAnchor),
            agendaTableView.bottomAnchor.constraint(equalTo: view.bottomAnchor),
        ])
    }

    @objc private func previousMonth() {
        let newMonth = offsetMonth(viewModel.currentMonth, by: -1)
        Task { await viewModel.load(month: newMonth) }
    }

    @objc private func nextMonth() {
        let newMonth = offsetMonth(viewModel.currentMonth, by: 1)
        Task { await viewModel.load(month: newMonth) }
    }

    private func offsetMonth(_ month: String, by value: Int) -> String {
        let formatter = DateFormatter()
        formatter.dateFormat = "yyyy-MM"
        guard let date = formatter.date(from: month) else { return month }
        let newDate = Calendar.current.date(byAdding: .month, value: value, to: date)!
        return formatter.string(from: newDate)
    }
}

extension CalendarViewController: UICollectionViewDataSource, UICollectionViewDelegate {
    func collectionView(_ collectionView: UICollectionView, numberOfItemsInSection section: Int) -> Int {
        daysInMonth.count
    }

    func collectionView(_ collectionView: UICollectionView, cellForItemAt indexPath: IndexPath) -> UICollectionViewCell {
        let cell = collectionView.dequeueReusableCell(withReuseIdentifier: "DayCell", for: indexPath) as! CalendarDayCell
        let date = daysInMonth[indexPath.item]
        let dayNumber = String(date.suffix(2))
        let hasEvents = !viewModel.chores(for: date).isEmpty
        cell.configure(day: dayNumber, hasEvents: hasEvents, isSelected: date == selectedDate)
        return cell
    }

    func collectionView(_ collectionView: UICollectionView, didSelectItemAt indexPath: IndexPath) {
        selectedDate = daysInMonth[indexPath.item]
        agendaChores = viewModel.chores(for: selectedDate ?? "")
        collectionView.reloadData()
        agendaTableView.reloadData()
    }
}

extension CalendarViewController: UITableViewDataSource {
    func tableView(_ tableView: UITableView, numberOfRowsInSection section: Int) -> Int {
        agendaChores.count
    }

    func tableView(_ tableView: UITableView, cellForRowAt indexPath: IndexPath) -> UITableViewCell {
        let cell = tableView.dequeueReusableCell(withIdentifier: "AgendaCell", for: indexPath)
        let chore = agendaChores[indexPath.row]
        cell.textLabel?.text = chore.name
        cell.detailTextLabel?.text = chore.status.rawValue
        return cell

}

    required init?(coder: NSCoder) { fatalError() }

    func configure(day: String, hasEvents: Bool, isSelected: Bool) {

        dotView.isHidden = !hasEvents
        contentView.backgroundColor = isSelected ? .systemBlue.withAlphaComponent(0.2) : .clear
        contentView.layer.cornerRadius = 4
    }
```
